Here's how to properly request microphone access:

// Function to request microphone access in response to user action
async function requestMicrophoneAccess() {
    try {
        // Request with clear constraints
        const stream = await navigator.mediaDevices.getUserMedia({
            audio: {
                echoCancellation: true,
                noiseSuppression: true,
                autoGainControl: true
            }
        });
        
        // Handle success
        handleMicrophoneSuccess(stream);
    } catch (error) {
        // Handle specific error types
        if (error.name === 'NotAllowedError') {
            // User or browser denied permission
            handlePermissionDenied();
        } else if (error.name === 'NotFoundError') {
            // No microphone available
            handleNoMicrophone();
        } else {
            // Other errors
            handleOtherMicrophoneError(error);
        }
    }
}
